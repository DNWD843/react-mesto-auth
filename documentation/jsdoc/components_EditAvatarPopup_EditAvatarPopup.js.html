import React, { useCallback } from 'react';
import PopupWithForm from '../PopupWithForm/PopupWithForm';

/**
 * @module EditAvatarPopup
 * @description Функциональный React-компонент<br>
 * Декларирует отрисовку попапа изменение аватара пользователя<br>
 * Принимает объект параметров props<br>
 * Возвращает JSX-фрагмент разметки попапа редактирования аватара пользователя
 * @param {Object} props - объект с параметрами
 * @param {Boolean} props.isOpen - маркер состояния попапа: true - открыт, false - закрыт
 * @param {Function} props.onClose - функция-коллбэк, закрывает попап
 * @param {Function} props.onOverlayClick - функция-коллбэк, вызывается при клике по оверлею, закрывает попап
 * @param {Function} props.onUpdateAvatar -  функция-коллбэк, передает ссылку на новый аватар в компонент App 
 * для обновления аватара пользователя
 * @param {Boolean} props.isLoading - маркер состояния загрузки нового аватара пользователя:
 * true - идет загрузка, false - нет загрузки
 * @returns {JSX} - JSX-фрагмент разметки, попап редактирования аватара пользователя
 * @see {@link PopupWithForm}
 * @since v.2.0.2
 */
function EditAvatarPopup(props) {

  /**
   * Реф инпута формы изменения аватара
   * @ignore
   */
  const inputRef = React.useRef();
  
  /**
   * @method handleSubmit
   * @description - Обработчик сабмита формы изменения аватара. 
   * Вызывает коллбэк props.onUpdateAvatar и передает ему ссылку на новый аватар пользователя 
   * @public
   * @since v.2.0.2
   */
  const handleSubmit = useCallback((evt) => {
    evt.preventDefault();
    props.onUpdateAvatar({
      avatar: inputRef.current.value
    });
  }, [props]);

  /**
   * При закрытии попапа очищаем инпут формы
   * @since v.2.0.2
   * @ignore
   */
  React.useEffect(() => {
    if (!props.isOpen) {
      inputRef.current.value = '';
    }
  }, [props.isOpen]);
    
  return (
    <PopupWithForm
      name="edit-avatar"
      title="Обновить аватар"
      submitButtonText="Создать"
      isOpen={ props.isOpen }
      onClose={ props.onClose }
      onOverlayClick={ props.onOverlayClick }
      onSubmit={ handleSubmit }
      isLoading={ props.isLoading }
      preloaderText="Сохранение..."
      children={
        <ul className="form__inputs">
          <li className="form__field">
            <input type="url" ref={ inputRef } className="form__input form__input_type_avatar-link" name="avatar-link-input" id="avatar-link-input" placeholder="Ссылка на картинку" required />
            <span className="form__input-error" id="avatar-link-input-error"></span>
          </li>
        </ul>
      }
    />
  );
}

export default EditAvatarPopup;
