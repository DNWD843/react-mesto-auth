import React, { useCallback } from 'react';
import PopupWithForm from '../PopupWithForm/PopupWithForm';
import { CurrentUserContext } from '../../contexts/CurrentUserContext';

/**
 * @module EditProfilePopup
 * @description Функциональный React-компонент<br>
 * Декларирует отрисовку и изменение данных профиля пользователя<br>
 * Принимает объект параметров props<br>
 * Возвращает JSX-фрагмент разметки попапа редактирования профиля пользователя
 * @param {Object} props - объект с параметрами
 * @param {Boolean} props.isOpen - маркер состояния попапа: true - открыт, false - закрыт
 * @param {Function} props.onClose - функция-коллбэк, закрывает попап
 * @param {Function} props.onOverlayClick - функция-коллбэк, вызывается при клике по оверлею, закрывает попап
 * @param {Function} props.onUpdateUser -  функция-коллбэк, передает параметры пользователя в компонент App 
 * для обновления профиля пользователя
 * @param {Boolean} props.isLoading - маркер состояния загрузки новых данных профиля пользователя: 
 * true - идет загрузка, false - нет загрузки
 * @returns {JSX} - JSX-фрагмент разметки, попап редактирования профиля пользователя
 * @see {@link PopupWithForm}
 * @since v.2.0.2
 */
function EditProfilePopup(props) {

  const currentUser = React.useContext(CurrentUserContext);
  const [formValues, setFormValues] = React.useState({
    name: '',
    description: ''
  });

  /**
   * @method handleInputChange
   * @description Обработчик изменения полей инпутов формы<br>
   * Сохраняет значения полей инпутов в стейтовые переменные  
   * @public
   * @since v.2.0.2
   */
  const handleInputChange = useCallback((e) => {
    const { name: inputName, value: inputValue } = e.target;
    setFormValues(prevValues => {
      return {
        ...prevValues,
        [inputName]: inputValue
      };
    });
  }, [setFormValues]);

  const { name, description } = formValues;

  /**
   * @method handleSubmit
   * @description Обработчик сабмита формы редактирования профиля. 
   * Вызывает коллбэк props.onUpdateUser и передает ему новые данные пользователя
   * @public
   * @since v.2.0.2
   */
  const handleSubmit = useCallback((evt) => {
    evt.preventDefault();
    props.onUpdateUser({
      name: formValues.name,
      about: formValues.description,
    });
  }, [props, formValues]);

  /**
   * После закрытия попапа устанавливаем актуальные данные профиля пользователя в инпутах формы
   * @since v.2.0.2
   * @ignore
   */
  React.useEffect(() => {
    if (!props.isOpen) {
      setFormValues({
        name: currentUser.name,
        description: currentUser.about
      });
    }
  }, [currentUser, props.isOpen]);

  return (
    <PopupWithForm
      name="edit-profile"
      title="Редактировать профиль"
      submitButtonText="Сохранить"
      isOpen={ props.isOpen }
      onClose={ props.onClose }
      onOverlayClick={ props.onOverlayClick }
      onSubmit={ handleSubmit }
      isLoading={props.isLoading }
      preloaderText="Сохранение..."
    >
      <>
        <ul className="form__inputs">
          <li className="form__field">
            <input type="text" onChange={ handleInputChange } className="form__input form__input_type_name" value={ name ? name : '' } name="name" id="user-name-input" placeholder="Имя" minLength="2" maxLength="40" required />
            <span className="form__input-error" id="user-name-input-error"></span>
          </li>
          <li className="form__field">
            <input type="text" onChange={ handleInputChange } className="form__input form__input_type_job" value={ description ? description : '' } name="description" id="user-job-input" placeholder="О себе" minLength="2" maxLength="200" required />
            <span className="form__input-error" id="user-job-input-error"></span>
          </li>
        </ul>
      </>
    </PopupWithForm>
  );
}

export default EditProfilePopup;
