import React, { useCallback } from 'react';
import PopupWithForm from '../PopupWithForm/PopupWithForm';

/**
 * @module AddPlacePopup
 * @description Функциональный React-компонент<br>
 * Декларирует отрисовку попапа добавления карточки<br>
 * Принимает объект параметров props<br>
 * Возвращает JSX-фрагмент разметки попапа добавления карточки
 * @param {Object} props - объект параметров
 * @param {Boolean} props.isOpen - маркер состояния попапа: true - открыт, false - закрыт
 * @param {Function} props.onClose - функция-коллбэк, закрывает попап
 * @param {Function} props.onOverlayClick - функция-коллбэк, вызывается при клике по оверлею, закрывает попап
 * @param {Function} props.onSubmit - функция-коллбэк сабмита формы добавления карточки
 * @param {Boolean} props.isLoading - маркер состояния загрузки новой карточки: true - идет загрузка, false - нет загрузки
 * @public
 * @see {@link PopupWithForm}
 * @since v.2.0.2
 */
function AddPlacePopup(props) {

  const [formValues, setFormValues] = React.useState({
    name: '',
    link: ''
  });

  /**
   * @method handleInputChange
   * @description Обработчик изменения полей инпутов формы<br>
   * Сохраняет значения полей инпутов в стейты
   * @public
   * @since v.2.0.2
   */
  const handleInputChange = useCallback((e) => {
    const { name: inputName, value: inputValue } = e.target;
    setFormValues(prevValues => {
      return {
        ...prevValues,
        [inputName]: inputValue
      };
    });
  }, [setFormValues]);

  const { name, link } = formValues;

  /**
   * @method handleSubmit
   * @description Обработчик сабмита формы добавления карточки. 
   * Вызывает коллбэк props.onSubmit и передает ему объект с данными полей формы
   * @public
   * @since v.2.0.2
   */
  const handleSubmit = useCallback((evt) => {
    evt.preventDefault();
    props.onSubmit({
      name: formValues.name,
      link: formValues.link
    });
  }, [props, formValues]);

  /**
   * Очистка полей формы добавления карточки после закрытия формы
   * @since v.2.0.2
   * @ignore
   */
  React.useEffect(() => {
    if (!props.isOpen) {
      setFormValues({
        name: '',
        link: ''
      });
    }
  }, [props.isOpen])

  return (
    <PopupWithForm
      name="add-photo"
      title="Новое место"
      submitButtonText="Создать" 
      isOpen={ props.isOpen }
      onClose={ props.onClose }
      onOverlayClick={ props.onOverlayClick }
      onSubmit={ handleSubmit }
      isLoading={ props.isLoading }
      preloaderText="Сохранение..."
    >      
      <>
        <ul className="form__inputs">
          <li className="form__field">
            <input type="text" onChange={handleInputChange} name="name" value={name ? name : ''} className="form__input form__input_type_place-title" id="place-title-input" minLength="1" maxLength="30" placeholder="Название" required />
            <span className="form__input-error" id="place-title-input-error"></span>
          </li>
          <li className="form__field">
            <input type="url" onChange={ handleInputChange } name="link" value={link ? link : ''} className="form__input form__input_type_image-link" id="image-link-input" placeholder="Ссылка на картинку" required />
            <span className="form__input-error" id="image-link-input-error"></span>
          </li>
        </ul>
      </>
    </PopupWithForm>
  )
}


export default AddPlacePopup;
